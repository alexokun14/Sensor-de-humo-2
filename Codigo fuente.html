#include 

const int sensorHumoPin = A0; 
const int ledPin = 8;     
int umbralHumo = 150;   

void setup() {
  pinMode(sensorHumoPin, INPUT); 
  pinMode(ledPin, OUTPUT);   

  vw_set_tx_pin(12);
  vw_setup(2000);  
}

void loop() {
  int valorHumo = analogRead(sensorHumoPin); 

  if (valorHumo > umbralHumo) { 
    digitalWrite(ledPin, HIGH);
    const char *mensaje = "Humo detectado";  
    vw_send((uint8_t *)mensaje, strlen(mensaje)); 
    vw_wait_tx(); 
  } else {
    digitalWrite(ledPin, LOW);
  }

  delay(1000); // Esperar 1 segundo antes de volver a medir
}

#include 

const int ledPin = 8;

void setup() {
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
  vw_set_rx_pin(11);
  vw_setup(2000);
  vw_rx_start(); 
}

void loop() {
  uint8_t buf[VW_MAX_MESSAGE_LEN];
  uint8_t buflen = VW_MAX_MESSAGE_LEN;

  if (vw_get_message(buf, &buflen)) {
    Serial.print("Mensaje recibido: ");
    for (int i = 0; i < buflen; i++) {
      Serial.print((char)buf[i]);
    }
    Serial.println();

    // Si el mensaje es "Humo detectado", encender el LED
    if (strncmp((char *)buf, "Humo detectado", buflen) == 0) {
      digitalWrite(ledPin, HIGH);
    }
  } else {
    digitalWrite(ledPin, LOW);
  }

  delay(100)
}
